### 20211004
- 순열 알고리즘
```js
function getPemutation(arr, selectedNum) {
    const result = [];
    if (selectedNum === 1) {
        return arr.map((v) => [v]);
    }

    arr.forEach((v, index) => {
        const rest = arr.filter((_, idx) => index !== idx);
        const restPemutations = getPemutation(rest, selectedNum - 1);
        const combine = restPemutations.map((restPemutation) => [v, ...restPemutation]);
        result.push(...combine)
    });

    return result;
}
```
